Install AgentMind using pip:

```bash
pip install agentmind
```

## Basic Usage

### Initialize Memory

```python
from agentmind import Memory

# Initialize with your API key
memory = Memory(api_key="am_live_YOUR_API_KEY")
```

### Store Memories

AgentMind accepts any data type - strings, dicts, lists. Use metadata for organization:

```python
# Store user-specific preference
memory.remember(
    "Prefers email over phone calls, send weekly summaries",
    user_id="user_123",
    metadata={"category": "preferences"}
)

# Store system configuration with metadata
memory.remember(
    {"rate_limit": 1000, "timeout": 30, "retry_count": 3},
    metadata={"category": "api_config", "service": "openai"}
)

# Store customer profile with user_id
memory.remember({
    "name": "John Doe",
    "tier": "enterprise", 
    "features": ["sso", "analytics", "priority_support"],
    "contract_value": 50000,
    "renewal_date": "2024-12-01"
}, user_id="user_123", metadata={"type": "profile"})
```

### Retrieve Memories

#### Natural Language Search

```python
# Search for user-specific memories
results = memory.recall(
    "How does user_123 want to be contacted?",
    metadata_filter={"user_id": "user_123"}
)
# Returns: ["Prefers email over phone calls, send weekly summaries"]

# Search for configuration
api_config = memory.recall(
    "OpenAI API configuration",
    metadata_filter={"category": "api_config", "service": "openai"}
)
# Returns the rate limit and timeout settings

# Search across all user issues
recent_issues = memory.recall(
    "unresolved customer problems",
    metadata_filter={"status": "open", "type": "issue"},
    limit=5
)
```

## Agent Tool Integration

The primary use case for AgentMind is giving your agents memory tools:

```python
# Define memory tools for your agent with user context
def create_memory_tools(user_id: str):
    memory = Memory(api_key="am_live_YOUR_API_KEY")
    
    def remember_tool(content: str, category: str = "general"):
        """Store information with user context."""
        return memory.remember(
            content=content,
            metadata={
                "user_id": user_id,
                "category": category,
                "timestamp": datetime.now().isoformat()
            }
        )
    
    def recall_tool(query: str, category: str = None):
        """Recall information for this user."""
        filters = {"user_id": user_id}
        if category:
            filters["category"] = category
            
        memories = memory.recall(query, metadata_filter=filters, limit=5)
        return memories if memories else []
    
    return remember_tool, recall_tool

# Create tools for a specific user
remember, recall = create_memory_tools("user_123")

# Add to your agent
agent = YourAgent(
    tools=[remember, recall],
    system_prompt="You have memory tools for this user. Remember important details."
)
```

## Complete Example

Here's a practical AI assistant personalization example:

```python
from agentmind import Memory
from datetime import datetime

# Initialize memory
memory = Memory(api_key="am_live_YOUR_API_KEY")

# AI Assistant learning user's work patterns
user_id = "user_dev_456"
session_id = "session_" + datetime.now().strftime("%Y%m%d_%H%M%S")

# Store coding preferences
memory.remember(
    "Prefers TypeScript with strict mode, uses React hooks over class components",
    metadata={
        "user_id": user_id,
        "category": "coding_preferences",
        "language": "typescript",
        "framework": "react"
    }
)

# Store work patterns
memory.remember({
    "peak_productivity": "9am-12pm",
    "preferred_session_length": "2 hours",
    "break_pattern": "pomodoro",
    "focus_music": "lo-fi beats"
}, metadata={
    "user_id": user_id,
    "category": "work_patterns",
    "last_updated": datetime.now().isoformat()
})

# Store frequently used tools and commands
memory.remember({
    "ide": "VSCode",
    "terminal": "iTerm2",
    "common_commands": ["npm run dev", "git status", "docker-compose up"],
    "snippets": ["rafce", "useState", "useEffect"]
}, metadata={
    "user_id": user_id,
    "category": "dev_environment"
})

# Later, AI assistant personalizes responses
coding_style = memory.recall(
    "TypeScript React preferences",
    metadata_filter={"user_id": user_id, "category": "coding_preferences"}
)
# AI generates code examples matching user's style

work_patterns = memory.recall(
    "productivity patterns",
    metadata_filter={"user_id": user_id, "category": "work_patterns"}
)
# AI suggests breaks and focus sessions based on patterns

# Track learning progress
memory.remember(
    "Learned about Next.js App Router, interested in server components",
    metadata={
        "user_id": user_id,
        "category": "learning_journey",
        "topic": "nextjs",
        "date": datetime.now().isoformat()
    }
)
```
